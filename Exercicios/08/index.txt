namespace CsharpAD06;

class Program
{
    static void Main(string[] args)
    {
        ApresentarPrograma();

        LacoRepeticao();
    }

    public static void ApresentarPrograma()
    {
        Console.WriteLine("-- EXERCÍCIOS DE LAÇO DE REPETIÇÃO --");
    }

    public static void LacoRepeticao()
    {
        Console.WriteLine("Determine o a base de exibição de asterisco(s):");
        int Base = Convert.ToInt32(Console.ReadLine());

        Console.WriteLine("Determine a altura de exibição de asterisco(s):");
        int Altura = Convert.ToInt32(Console.ReadLine());

        for (int i = 1; i <= Altura; i++)
        {
            for (int j = 1; j <= Base; j++)
            {
                Console.Write('*');
            }
            Console.WriteLine();
        }
    }
}
